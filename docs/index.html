// quick start

shell npm install @sonata-sdk/decoder @sonata-sdk/ws @sonata-sdk/voice
typescript import { decodeMP3 } from '@sonata-sdk/decoder' import { readFileSync } from 'node:fs' const buf = readFileSync('track.mp3') const { channelData, sampleRate } = await decodeMP3(buf) // channelData[0] → Float32Array left channel
typescript import { ResumableWS } from '@sonata-sdk/ws' const ws = new ResumableWS('wss://gateway.discord.gg') ws.on('open', () => console.log('connected')) await ws.connect()
typescript import { joinVoiceChannel } from '@sonata-sdk/voice' const conn = joinVoiceChannel({ guildId: '123', userId: '456', encryption: 'aead_aes256_gcm_rtpsize', }) conn.on('ready', () => { conn.sendAudioFrame(opusFrame) })
